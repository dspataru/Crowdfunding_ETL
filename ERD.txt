# Course: UTOR-VIRT-DATA-PT-06-2023-U-LOLC
# Authors: Daiana Spataru, Jasmine Bamba
# Context: Create ERD for the Crowdfunding Database


Contacts
-
contact_id PK INT
first_name VARCHAR(50)
last_name VARCHAR(50)
email VARCHAR(100)


Campaign
-
cf_id PK INT
contact_id INT FK - Contacts.contact_id
company VARCHAR(250)
description VARCHAR
goal FLOAT
pledged FLOAT
outcome VARCHAR(10)
backers_count INT
country VARCHAR(2)
currency VARCHAR(3)
launched_date DATE
end_date DATE
category_id INT FK - Category.category_id
subcategory_id INT FK - Subcategory.subcategory_id


Subcategory
----
subcategory_id PK INT
subcategory VARCHAR(50)


Category
----
category_id PK INT
category VARCHAR(50)
